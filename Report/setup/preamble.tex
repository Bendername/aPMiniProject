%PIE CHART FUNCTION
\newcommand{\slice}[5]{
  \pgfmathparse{0.5*#2+0.5*#3}
  \let\midangle\pgfmathresult

  % slice
   \pgfmathparse{0.5*rnd+0.3}
   \let\randomrgb\pgfmathresult

   \draw[thick, fill=#1] (0,0) -- (#2:1) arc (#2:#3:1) -- cycle;

  % outer label
  \node[label=\midangle:#5] at (\midangle:1) {};

  % inner label
  \pgfmathparse{min((#3-#2-10)/110*(-0.3),0)}
  \let\temp\pgfmathresult
  \pgfmathparse{max(\temp,-0.5) + 0.8}
  \let\innerpos\pgfmathresult
  \node at (\midangle:\innerpos) {#4};
}
